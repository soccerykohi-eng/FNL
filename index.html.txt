        // AUTOMATIC STANDINGS ENGINE
        function calculateStandings() {
            const standingsMap = {};

            appState.teams.forEach(t => {
                standingsMap[t.name] = {
                    name: t.name,
                    tag: t.tag,
                    logo: t.logo,
                    captain: t.captain,
                    p: 0, w: 0, d: 0, l: 0, gf: 0, ga: 0, gd: 0, pts: 0,
                    form: []
                };
            });

            appState.matches.forEach(m => {
                if (m.status === 'finished' && m.homeScore !== null && m.awayScore !== null) {
                    const home = standingsMap[m.home];
                    const away = standingsMap[m.away];

                    if (home && away) {
                        home.p++;
                        away.p++;

                        home.gf += m.homeScore;
                        home.ga += m.awayScore;
                        away.gf += m.awayScore;
                        away.ga += m.homeScore;

                        if (m.homeScore > m.awayScore) {
                            home.w++;
                            home.pts += 3;
                            away.l++;
                            home.form.push('W');
                            away.form.push('L');
                        } else if (m.homeScore < m.awayScore) {
                            away.w++;
                            away.pts += 3;
                            home.l++;
                            away.form.push('W');
                            home.form.push('L');
                        } else {
                            home.d++;
                            away.d++;
                            home.pts += 1;
                            away.pts += 1;
                            home.form.push('D');
                            away.form.push('D');
                        }
                    }
                }
            });

            const standingsList = Object.values(standingsMap).map(t => {
                t.gd = t.gf - t.ga;
                return t;
            });

            standingsList.sort((a, b) => {
                if (b.pts !== a.pts) return b.pts - a.pts;
                if (b.gd !== a.gd) return b.gd - a.gd;
                if (b.gf !== a.gf) return b.gf - a.gf;
                return b.w - a.w;
            });

            standingsList.forEach((t, idx) => {
                t.rank = idx + 1;
            });

            return standingsList;
        }

        }

        // SAVE MATCH SCORE
        function saveMatchScore(matchId) {
            const m = appState.matches.find(item => item.id === matchId);
            if (!m) return;

            const homeVal = parseInt(document.getElementById('edit-home-score').value, 10);
            const awayVal = parseInt(document.getElementById('edit-away-score').value, 10);
            const statusVal = document.getElementById('edit-match-status').value;

            m.homeScore = isNaN(homeVal) ? 0 : homeVal;
            m.awayScore = isNaN(awayVal) ? 0 : awayVal;
            m.status = statusVal;

            saveState();
            renderStandings('total');
            renderMatches(appState.selectedMwFilter);
            renderFeaturedMatch();
            closeSheetDirect();
        }

        // ADD NEWS
        function addNewsPost() {
            const titleInput = document.getElementById('admin-news-title');
            if (!titleInput.value.trim()) return;

            const now = new Date();
            const dateStr = `${now.getFullYear()}.${String(now.getMonth() + 1).padStart(2, '0')}.${String(now.getDate()).padStart(2, '0')}`;

            appState.news.unshift({
                date: dateStr,
                title: titleInput.value.trim()
            });

            titleInput.value = '';
            saveState();
            renderNews();
            alert('お知らせを投稿しました！');
        }

        // ADD SCORER
        function addScorerGoal() {
            const nameInput = document.getElementById('admin-scorer-name');
            const teamSelect = document.getElementById('admin-scorer-team');
            const goalsInput = document.getElementById('admin-scorer-goals');

            if (!nameInput.value.trim()) return;

            const scorerName = nameInput.value.trim();
            const teamName = teamSelect.value;
            const addGoals = parseInt(goalsInput.value, 10) || 1;

            const existing = appState.scorers.find(s => s.name === scorerName && s.team === teamName);
            if (existing) {
                existing.goals += addGoals;
            } else {
                appState.scorers.push({
                    id: Date.now(),
                    name: scorerName,
                    team: teamName,
                    goals: addGoals
                });
            }

            nameInput.value = '';
            saveState();
            renderStandings('scorers');
            alert('得点記録を更新しました！');
        }

        function populateTeamDropdown() {
            const select = document.getElementById('admin-scorer-team');
            if (!select) return;
            select.innerHTML = '';
            appState.teams.forEach(t => {
                const opt = document.createElement('option');
                opt.value = t.name;
                opt.innerText = t.name;
                select.appendChild(opt);
            });
        }

        function closeSheet(e) {
            if (e.target.id === 'sheet-overlay') {
                closeSheetDirect();
            }
        }

        function closeSheetDirect() {
            document.getElementById('sheet-overlay').classList.remove('active');
        }

        function switchStandingsFilter(type, el) {
            el.parentElement.querySelectorAll('.chip').forEach(c => c.classList.remove('active'));
            el.classList.add('active');
            renderStandings(type);
        }

        function filterMatches(sec, el) {
            el.parentElement.querySelectorAll('.chip').forEach(c => c.classList.remove('active'));
            el.classList.add('active');
            renderMatches(sec);
        }
